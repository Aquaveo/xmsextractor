106 double iGetEarcutTriangleRatio(const VecPt3d& a_points, int a_idx1, int a_idx2, int a_idx3,
107  RatioCache& a_ratioCache)
108 {
109  Triangle triangle(a_idx1, a_idx2, a_idx3);
110  auto ratioIter = a_ratioCache.find(triangle);
111  if (ratioIter != a_ratioCache.end())
112  return ratioIter->second;
113 
114  Pt3d pt1 = a_points[a_idx1];
115  Pt3d pt2 = a_points[a_idx2];
116  Pt3d pt3 = a_points[a_idx3];
117  Pt3d v1 = pt1 - pt2;
118  Pt3d v2 = pt3 - pt2;
119  Pt3d v3 = pt3 - pt1;
120 
121  // get 2*area
122  Pt3d cross;
123  gmCross3D(v2, v1, &cross);
124  double ratio;
125  if (cross.z <= 0.0)
126  {
127  // inverted
128  ratio = -1.0;
129  }
130  else
131  {
132  double area2 = iMagnitude(cross);
133  if (area2 == 0.0)
134  {
135  // degenerate triangle
136  ratio = -2.0;
137  }
138  else
139  {
140  double perimeter = iMagnitude(v1) + iMagnitude(v2) + iMagnitude(v3);
141  ratio = perimeter * perimeter / area2;
142  }
143  }
144 
145  a_ratioCache[triangle] = ratio;
146  return ratio;
147 } // iGetEarcutTriangleRatio

151 bool iValidTriangle(const VecPt3d& a_points,
152  const VecInt a_polygon,
153  int a_idx1,
154  int a_idx2,
155  int a_idx3,
156  ValidCache& a_validCache)
157 {
158  Triangle triangle(a_idx1, a_idx2, a_idx3);
159  auto validIter = a_validCache.find(triangle);
160  if (validIter != a_validCache.end())
161  return validIter->second;
162 
163  const Pt3d& pt1 = a_points[a_idx1];
164  const Pt3d& pt2 = a_points[a_idx2];
165  const Pt3d& pt3 = a_points[a_idx3];
166  for (size_t pointIdx = 0; pointIdx < a_polygon.size(); ++pointIdx)
167  {
168  int idx = a_polygon[pointIdx];
169  if (idx != a_idx1 && idx != a_idx2 && idx != a_idx3)
170  {
171  const Pt3d& pt = a_points[idx];
172  if (gmTurn(pt1, pt2, pt, 0.0) == TURN_LEFT && gmTurn(pt2, pt3, pt) == TURN_LEFT &&
173  gmTurn(pt3, pt1, pt, 0.0) == TURN_LEFT)
174  {
175  a_validCache[triangle] = false;
176  return false;
177  }
178  }
179  }
180 
181  a_validCache[triangle] = true;
182  return true;
183 } // iValidTriangle

192 bool iGenerateCentroidTriangles(XmUGridTriangles2dImpl& a_ugridTris,
193  int a_cellIdx,
194  const VecInt& a_polygonIdxs)
195 {
196  const VecPt3d& points = a_ugridTris.GetPoints();
197  size_t numPoints = a_polygonIdxs.size();
198 
199  VecPt3d polygon;
200  for (size_t pointIdx = 0; pointIdx < numPoints; ++pointIdx)
201  polygon.push_back(points[a_polygonIdxs[pointIdx]]);
202 
203  Pt3d centroid = gmComputeCentroid(polygon);
204 
205  // make sure none of the triangles connected to the centroid are inverted
206  for (size_t pointIdx = 0; pointIdx < polygon.size(); ++pointIdx)
207  {
208  const Pt3d& pt1 = polygon[pointIdx];
209  const Pt3d& pt2 = polygon[(pointIdx + 1) % numPoints];
210 
211  // centroid should be to the left of each edge
212  if (gmTurn(pt1, pt2, centroid, 0.0) != TURN_LEFT)
213  return false;
214  }
215 
216  // add centroid to list of points
217  int centroidIdx = a_ugridTris.AddCellCentroid(a_cellIdx, centroid);
218 
219  // add triangles
220  for (size_t pointIdx = 0; pointIdx < polygon.size(); ++pointIdx)
221  {
222  int idx1 = a_polygonIdxs[pointIdx];
223  int idx2 = a_polygonIdxs[(pointIdx + 1) % numPoints];
224  a_ugridTris.AddCellTriangle(a_cellIdx, idx1, idx2, centroidIdx);
225  }
226 
227  return true;
228 } // iGenerateCentroidTriangles

235 void iBuildEarcutTriangles(XmUGridTriangles2dImpl& a_ugridTris,
236  int a_cellIdx,
237  const VecInt& a_polygonIdxs)
238 {
239  VecInt polygonIdxs = a_polygonIdxs;
240  const VecPt3d& points = a_ugridTris.GetPoints();
241 
242  // continually find best triangle on adjacent edges and cut it off polygon
243  RatioCache ratioCache;
244  ValidCache validCache;
245  while (polygonIdxs.size() >= 4)
246  {
247  int bestIdx = -1;
248  int secondBestIdx = -1;
249 
250  int numPoints = (int)polygonIdxs.size();
251  double bestRatio = std::numeric_limits<double>::max();
252  double secondBestRatio = std::numeric_limits<double>::max();
253  for (int pointIdx = 0; pointIdx < numPoints; ++pointIdx)
254  {
255  int idx1 = polygonIdxs[(pointIdx + numPoints - 1) % numPoints];
256  int idx2 = polygonIdxs[pointIdx];
257  int idx3 = polygonIdxs[(pointIdx + 1) % numPoints];
258 
259  // make sure triangle is valid (not inverted and doesn't have other points in it)
260  double ratio = iGetEarcutTriangleRatio(points, idx1, idx2, idx3, ratioCache);
261  if (ratio > 0.0)
262  {
263  if (iValidTriangle(points, polygonIdxs, idx1, idx2, idx3, validCache))
264  {
265  if (ratio < bestRatio)
266  {
267  secondBestRatio = bestRatio;
268  bestRatio = ratio;
269  secondBestIdx = bestIdx;
270  bestIdx = pointIdx;
271  }
272  else if (ratio < secondBestRatio)
273  {
274  secondBestRatio = ratio;
275  secondBestIdx = pointIdx;
276  }
277  }
278  }
279  }
280 
281  if (bestIdx >= 0)
282  {
283  if (numPoints == 4 && secondBestIdx >= 0)
284  bestIdx = secondBestIdx;
285 
286  // cut off the ear triangle
287  int polygonIdx1 = (bestIdx + numPoints - 1) % numPoints;
288  int polygonIdx2 = bestIdx;
289  int polygonIdx3 = (bestIdx + 1) % numPoints;
290  int idx1 = polygonIdxs[polygonIdx1];
291  int idx2 = polygonIdxs[polygonIdx2];
292  int idx3 = polygonIdxs[polygonIdx3];
293  a_ugridTris.AddCellTriangle(a_cellIdx, idx1, idx2, idx3);
294  polygonIdxs.erase(polygonIdxs.begin() + polygonIdx2);
295  }
296  else
297  {
298  std::ostringstream ss;
299  ss << "Unable to split cell number " << a_cellIdx + 1 << " into triangles.";
300  XM_LOG(xmlog::error, ss.str());
301  return;
302  }
303  }
304 
305  // push on remaining triangle
306  a_ugridTris.AddCellTriangle(a_cellIdx, polygonIdxs[0], polygonIdxs[1], polygonIdxs[2]);
307 } // iBuildEarcutTriangles
